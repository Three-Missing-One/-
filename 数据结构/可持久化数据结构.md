# 可持久化数据结构

## 可持久化线段树

- 函数式线段树，可以保留线段树的历史版本，不是一颗完全二叉树，必须记录节点的左右编号
- 可以不在记录每个节点代表的区间，而是作为参数递归向下查询
- 支持单点修改，区间查询，单点查询，不支持区间修改


### 经典例题
- [区间第K大](https://www.acwing.com/problem/content/257/)

题意：给一个序列，然后m次查询，每次询问l,r,k意味者查询l到r区间内第k大的数，我们按照序列顺序插入节点，得到插入每一个元素时的可持久版本，然后查询的l到r就是查询插入l-1时可持久版本k到r可持久版本的区别（使用权值线段树存的是下标），然后在递归查询

前置知识：
- 动态开点
- 权值线段树

```cpp
#include <bits/stdc++.h>
using namespace std;
const int maxn = 1e5+5;
int n,m;
int a[maxn];
int H[maxn];
struct SegmentTree{
    int lc,rc;    //左右孩子编号
    int cnt;     //区间内的数量
}tree[maxn<<5];
int root[maxn];
int tot = 0;
void pushup(int pos){
    tree[pos].cnt = tree[tree[pos].lc].cnt + tree[tree[pos].rc].cnt;
}
int build(int l,int r){
    int p = tot++;
    if(l==r){tree[p].cnt = 0;return  p;}
    int mid = (l+r)>>1;
    tree[p].lc = build(l,mid);
    tree[p].rc = build(mid + 1,r);
    pushup(p);
    return p;
}
int insert(int now,int l,int r,int x,int v){
    int p = tot++;
    tree[p] = tree[now];
    if(l == r){
        tree[p].cnt += v;
        return p;
    }
    int mid = (l+r)>>1;
    if(x <= mid)tree[p].lc = insert(tree[now].lc,l,mid,x,v);
    else tree[p].rc = insert(tree[now].rc,mid+1,r,x,v);
    pushup(p);
    return p;
}
int query(int u,int v,int l,int r,int k){//lr代表维护区间，uv代表的是可持久化版本

    int mid = (l + r)>>1;
    int x = tree[tree[v].lc].cnt - tree[tree[u].lc].cnt;
    if(l==r){
        return l;
    } 
    if(x>=k)return query(tree[u].lc,tree[v].lc,l,mid,k);
    else return query(tree[u].rc,tree[v].rc,mid+1,r,k-x);
}
int main(){
    scanf("%d%d",&n,&m);
    int cnt = 1;
    for(int i = 1; i <= n; i++){
        scanf("%d",&a[i]);
        H[cnt++] = a[i];
    }
    sort(H+1,H+cnt);
    int nn = unique(H+1,H+cnt)-(H+1);
    for(int i = 1;i <= n ;i ++){
        a[i] = upper_bound(H+1,H+nn+1,a[i]) - (H+1);
        // cout<<a[i]<<"#";
    }
    root[0] = build(1,1e5);
    for(int i = 1;i <= n; i++){
        root[i] = insert(root[i-1],1,1e5,a[i],1);
    }
    for(int i = 0 ;i < m; i++){
        int l,r,k;
        scanf("%d%d%d",&l,&r,&k);
        int q = query(root[l-1],root[r],1,1e5,k);
        printf("%d\n",H[q]);
    }
    

}
```

- [牛客多校2021-第九场E](https://ac.nowcoder.com/acm/contest/11260/E)

题意：有一棵树，然后每个节点有一个温度，然后每个我们每次询问给出一个节点，然后一个温度区间，我们从给出的节点出发，然后遍历在指定温度内的节点，问可以扩展几个节点

思路还是挺明确的，大致就是dfs先扫一遍，建一个dfs序，把树上问题转换成区间问题，优于在首都的点（1）号结点的温度最高，我们应该沿当前节点向上找一个延伸到最后的节点（树上向上倍增），然后在从最上面那个点向下查询l，我们可以需要查询在指定下表区间内数值范围的点，我们可以使用可持久化线段树，然后指定下表就是插入第i次和插入第j次区间的值的数量，然后我们在这个基础上查询区间（就是l到r的值，指定有几个）就好了
比赛时后没有判断条件mle了，以后一定注意边界条件，然后多写点这种题目。

```cpp
#include<bits/stdc++.h>
#define INF 0x3f3f3f3f
using namespace std;
const int maxn = 1e5+5;
int dfn[maxn];
int bk = 0;
int n;
struct edge{
    int v,next;
}e[maxn<<1];
int cnt = 0;
int head[maxn];
int t[maxn];
int H[maxn];
int sz[maxn];
void add(int u,int v){
    e[cnt].v = v;
    e[cnt].next = head[u];
    head[u] = cnt++;
}
int num = 1;
int df[maxn];
void dfs(int x,int fa){
    // dfn.push_back(x);
    dfn[bk++] = x;
    df[x] = num++;
    sz[x] = 1;
    for(int i = head[x];~i; i= e[i].next){
        int v = e[i].v;
        if(fa==v)continue;
        dfs(v,x);
        sz[x] += sz[v];
    }    
}
bool d[maxn];
queue<int>q;
int f[maxn][30];
void bfs(){
    q.push(1);d[1] = true;
    while(!q.empty()){
        int x = q.front();q.pop();
        for(int i = head[x];~i;i = e[i].next){
            int v = e[i].v;
            if(d[v])continue;
            d[v] = 1;
            f[v][0] = x;
            int T = log2(n)+1;
            for(int j = 1;j <= T;j++){
                f[v][j] = f[f[v][j-1]][j-1];
            }
            q.push(v);
        }
    }
}
int findr(int u,int r){
    int T = log2(n)+1;
    for(int i = T;i >= 0;i--){
        if(H[t[f[u][i]]] <= r)u = f[u][i];
    }
    return u;
}
struct SegmentTree{
    int lc,rc;    //左右孩子编号
    int cnt;     //区间内的数量
}tree[maxn<<5];
int root[maxn];
int tot = 0;

void pushup(int pos){
    tree[pos].cnt = tree[tree[pos].lc].cnt + tree[tree[pos].rc].cnt;
}
int build(int l,int r){
    int p = tot++;
    if(l==r){tree[p].cnt = 0;return  p;}
    int mid = (l+r)>>1;
    tree[p].lc = build(l,mid);
    tree[p].rc = build(mid + 1,r);
    pushup(p);
    return p;
}
int insert(int now,int l,int r,int x,int v){
    int p = tot++;
    tree[p] = tree[now];
    if(l == r){
        tree[p].cnt += v;
        return p;
    }
    int mid = (l+r)>>1;
    if(x <= mid)tree[p].lc = insert(tree[now].lc,l,mid,x,v);
    else tree[p].rc = insert(tree[now].rc,mid+1,r,x,v);
    pushup(p);
    return p;
}

int query(int u,int v,int l,int r,int ll,int rr){
    if(l>=r){
        return tree[v].cnt-tree[u].cnt;
    }
    if(ll<=H[l]&&rr>=H[r]){
        return tree[v].cnt-tree[u].cnt;
    }
    int mid = (l + r)>>1;
    int ans = 0;
    if(ll<=H[mid])ans += query(tree[u].lc,tree[v].lc,l,mid,ll,rr);
    if(rr>H[mid])ans += query(tree[u].rc,tree[v].rc,mid+1,r,ll,rr);
    return ans ;
}

int main(){
    memset(head,-1,sizeof(head));
    scanf("%d",&n);
    for(int i = 0 ;i< n-1; i++){
        int u,v;
        scanf("%d%d",&u,&v);
        add(u,v);add(v,u);
    }
    dfs(1,0);
    bfs();
    int ct = 0; 
    for(int i = 1;i<=n;i++){
        scanf("%d",&t[i]);
        H[++ct] = t[i];
    }
    sort(H+1,H+ct+1);
    int nn = unique(H+1,H+ct+1) - (H+1);
    for(int i = 1;i <= n; i++){
        t[i] = lower_bound(H+1,H+nn+1,t[i]) - (H);
    }
    H[0] = INF;
    root[0] = build(1,nn);
    for(int i = 0;i < bk;i++){
        root[i+1] = insert(root[i],1,nn,t[dfn[i]],1);
        // cout<<root[i+1]<<"#"<<i+1<<endl;
    }
    int Q;
    scanf("%d",&Q);
    for(int i = 0;i < Q ;i++){

        }
```
