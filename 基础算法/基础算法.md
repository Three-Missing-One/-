## 二分模板
```cpp
//ans>=mi,ans<=ma
l=mi,r=ma,mid;
while(l<=r)
{
    mid=(l+r)>>1;
    if(isOk(mid))
    {
        //求最大值
        l=mid+1;
        ans=max(ans,mid);
        //求最小值
        r=mid-1;
        ans=min(ans,mid);
    }
    else
    {
        //求最大值
        r=mid-1;
        //求最小值
        l=mid+1;
    }
}
```
## 尺取模板

```cpp
//ans:答案，n：个数，t给定标准 
void ruler(){
    int l=0,r=0,sum=0;//区间[0，n)
    int ans=0;
    while(1){
        while(r<n&&sum+a[r]<=t){//因为右边是开区间，用r试探，如果下一个满
        //足加起来小于等于t，sum就加它，
            sum+=a[r++];
        }
        ans=max(ans,r-l);
        if(r==n)break;//区间[0，n)
        sum-=a[l++];//l移动
    }
    printf("%d\n",ans);
}
```
